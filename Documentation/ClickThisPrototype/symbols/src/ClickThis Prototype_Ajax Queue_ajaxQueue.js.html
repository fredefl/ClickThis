 12 /**
 13 * ajaxQueue Class
 14 * @class ajaxQueue Class
 15 */
 16 "use strict";
 17 var ajaxQueue = {
 18 	/**
 19 	 * The key name for the localStorage row
 20 	 * @type {string}
 21 	 */
 22 	localStorageKeyName: "ajaxQueue",
 23 
 24 	/**
 25 	 * The length of the queue element id
 26 	 * @type {integer}
 27 	 */
 28 	idLength: 4,
 29 
 30 	/**
 31 	 * The queue array
 32 	 * @type {array}
 33 	 */
 34 	queueArray: null,
 35 
 36 	/**
 37 	 * The array the holds the function(s) to callback to
 38 	 * @type {array}
 39 	 */
 40 	callbackArray: {},
 41 
 42 	/**
 43 	 * Logs the specified message
 44 	 * @param  {string} message The message to log
 45 	 * @return {null}
 46 	 */
 47 	log: function (message) {
 48 		$('#log').append(message + '\r\n');
 49 	},
 50 
 51 	/**
 52 	 * Loads the queue from localStorage
 53 	 * @return {boolean}
 54 	 */
 55 	load: function () {
 56 		if (localStorage.getItem(this.localStorageKeyName) !== null) {
 57 			var localStorageItem = localStorage.getItem(this.localStorageKeyName),
 58 				jsonArray = JSON.parse(localStorageItem);
 59 			this.queueArray = jsonArray;
 60 			ajaxQueue.log("Data loaded from localstorage, added " + jsonArray.Tasks.length + " items to queue.");/*LOG*/
 61 			return true;
 62 		} else {
 63 			this.queueArray = {Tasks: []};
 64 			ajaxQueue.log("No data in localstorage, adding new key.");/*LOG*/
 65 			return false;
 66 		}
 67 	},
 68 
 69 	/**
 70 	 * Saves the queue array to localStorage
 71 	 * @return {boolean}
 72 	 */
 73 	save: function () {
 74 		var jsonString = JSON.stringify(this.queueArray, null, 2);
 75 		localStorage.setItem(this.localStorageKeyName, jsonString);
 76 		ajaxQueue.log("Saved queue to localstorage.");/*LOG*/
 77 		return true;
 78 	},
 79 
 80 	/**
 81 	 * Adds an element to the queue array
 82 	 *
 83 	 * @param {string} url The url to send the ajax request to
 84 	 * @param {data} data The data to send with the ajax request (POST)
 85 	 * @param {data} group The group to add the data to
 86 	 * @returns {string} The new id of the element
 87 	 */
 88 	add: function (url, data, group) {
 89 		// Create a random string
 90 		var id = this.randomString(this.idLength);
 91 		// Insert the task into the queue
 92 		this.queueArray.Tasks.push({id: id, url: url, data: data, group: group});
 93 		// Save the queue to prevent data loss
 94 		this.save();
 95 		ajaxQueue.log("Added task, url:" + url + ", data:" + data);/*LOG*/
 96 		// Return the new id for the task
 97 		return id;
 98 	},
 99 
100 	/**
101 	 * Removes the specified task from the queue
102 	 */
103 	remove: function (id) {
104 		var result = false;
105 		$(this.queueArray.Tasks).each(function (index, element) {
106             if (element.id === id) {
107 				ajaxQueue.queueArray.Tasks.splice(index, 1);
108 				result = true;
109 			}
110         });
111 		this.save();
112 		ajaxQueue.log("Removed " + id);/*LOG*/
113 		return result;
114 	},
115 
116 	/**
117 	 * Clears the queue
118 	 */
119 	clear: function () {
120 		this.queueArray = {};
121 		this.save();
122 		ajaxQueue.log("Cleared queue");/*LOG*/
123 		return true;
124 	},
125 
126 	/**
127 	 * Executes the tasks in the queue
128 	 */
129 	executeTasks: function () {
130 		if (this.queueArray.Tasks.length > 0) {
131 			var currentTask = this.queueArray.Tasks[0],
132 				callback = null;
133 			ajaxQueue.log("Sending '" + currentTask.data + "' to '" + currentTask.url + "'.");/*LOG*/
134 			$.post(
135 				currentTask.url,
136 				currentTask.data,
137 				// On success
138 				function () {
139 					// Remove the task from the queue
140 					ajaxQueue.remove(currentTask.id);
141 					ajaxQueue.log("Sending of '" + currentTask.data + "' to '" + currentTask.url + "' was successfull.");/*LOG*/
142 					// Call the groups callback
143 					callback = ajaxQueue.callbackArray;
144 					if (callback && typeof(callback) === "function") {
145 						ajaxQueue.log("Calling back!");
146 						callback();
147 					}
148 					// Loop on...
149 					ajaxQueue.executeTasks();
150 				}
151 			);
152 		}
153 	},
154 
155 	/**
156 	 * Registers a callback function
157 	 * @param  {Function} callback The function to call
158 	 * @param  {string}   group    The group
159 	 * @return {boolean}
160 	 */
161 	registerCallback: function (callback, group) {
162 		if (callback && typeof(callback) === "function") {
163 			ajaxQueue.callbackArray[group] = callback;
164 			return true;
165 		} else {
166 			return false;
167 		}
168 	},
169 
170 	/**
171 	 * Generates a random id
172 	 * @return {string}
173 	 */
174 	generateId: function () {
175 		return ajaxQueue.randomString(5);
176 	},
177 
178 	/**
179 	 * Generates a random string at the specified length
180 	 *
181 	 * @param {integer} length The length of the string
182 	 * @returns {string} The random string
183 	 */
184 	randomString: function (length) {
185 		var i = 0,
186 			// Define the chars used
187 			chars = "0123456789ABCDEFGHIJKLMNOPQRSTUVWXTZabcdefghiklmnopqrstuvwxyz",
188 			// Initialize variables
189 			randomString = '',
190 			randomNumber = 0;
191 		// Loop *length* times
192 		for (i = 0; i < length; i++) {
193 			// Generate a random number
194 			randomNumber = Math.floor(Math.random() * chars.length);
195 			// Add a char to the random string
196 			randomString += chars.substring(randomNumber, randomNumber + 1);
197 		}
198 		return randomString;
199 	},
200 
201 	/**
202 	 * Removes an element from an array
203 	 * 
204 	 * @param {array} array The array to remove the element from
205 	 * @param {string} arrayElementName The name of the element to remove
206 	 */
207 	removeElementFromArray: function (array, arrayElementName) {
208 		var i = 0;
209 		for (i = 0; i < array.length; i++) {
210 			if (array[i] === arrayElementName) {
211 				array.splice(i, 1);
212 			}
213 		}
214 	}
215 };
