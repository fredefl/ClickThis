113 	/**
114 	 * Loads the queue from localStorage
115 	 * 
116 	 * @static
117 	 * @return {boolean} True if data was loaded from localStorage, false if no data was found in localStorage, and a new queue has been created.
118 	 * @example
119 	 * ajaxQueue.load();
120 	 */
121 	load: function () {
122 		if (localStorage.getItem(this.localStorageKeyName) !== null) {
123 			var localStorageItem = localStorage.getItem(this.localStorageKeyName),
124 				jsonArray = JSON.parse(localStorageItem);
125 			this.queueArray = jsonArray;
126 			ajaxQueue.log("Data loaded from localstorage, added " + jsonArray.Tasks.length + " items to queue.");/*LOG*/
127 			return true;
128 		} else {
129 			this.queueArray = {Tasks: []};
130 			ajaxQueue.log("No data in localstorage, adding new key.");/*LOG*/
131 			return false;
132 		}
133 		ajaxQueue.checkStatusCode();
134 	},
135 
136 	/**
137 	 * Saves the queue array to localStorage
138 	 * 
139 	 * @static
140 	 * @example
141 	 * ajaxQueue.save();
142 	 */
143 	save: function () {
144 		var jsonString = JSON.stringify(this.queueArray, null, 2);
145 		localStorage.setItem(this.localStorageKeyName, jsonString);
146 		ajaxQueue.log("Saved queue to localstorage.");/*LOG*/
147 		return true;
148 	},

150 	/**
151 	 * Adds an element to the queue array
152 	 *
153 	 * @static
154 	 * @param {String} json The json element to add.
155 	 * @returns {String} The new id of the element, false in case of error
156 	 * @example
157 	 * ajaxQueue.add({
158 	 *  url: "http://illution.dk",
159 	 *  data: "test=hehe,llama=fish",
160 	 *  group: "testGroup2"
161 	 * })
162 	 */
163 	add: function (json) {
164 		var id = null;
165 		if(json.data === undefined) {
166 			json.data = "";
167 		}
168 		if (json.url && json.group) {
169 			// Create a random string
170 			id = ajaxQueue.generateId();
171 			// Insert the task into the queue
172 			this.queueArray.Tasks.push({id: id, url: json.url, data: json.data, group: json.group});
173 			// Save the queue to prevent data loss
174 			this.save();
175 			ajaxQueue.log("Added task, url:" + json.url + ", data:" + json.data);/*LOG*/
176 		} else {
177 			// Incate that an error occoured
178 			id = false;
179 		}
180 		ajaxQueue.checkStatusCode();
181 		// Return the new id for the task
182 		return id;
183 	},
184 
185 	/**
186 	 * Removes the specified task from the queue
187 	 * 
188 	 * @static
189 	 * @param {String} id The id of the task to remove
190 	 * @example
191 	 * ajaxQueue.remove("5kB6");
192 	 */
193 	remove: function (id) {
194 		var result = false;
195 		$(this.queueArray.Tasks).each(function (index, element) {
196             if (element.id === id) {
197 				ajaxQueue.queueArray.Tasks.splice(index, 1);
198 				result = true;
199 			}
200         });
201 		this.save();
202 		ajaxQueue.log("Removed " + id);/*LOG*/
203 		ajaxQueue.checkStatusCode();
204 		return result;
205 	},

217 		ajaxQueue.log("Cleared queue");/*LOG*/
218 		ajaxQueue.checkStatusCode();
219 		return true;
220 	},

250 	 * @example
251 	 * ajaxQueue.getStatusCode()

254 	getStatusCode: function () {
255 		ajaxQueue.checkStatusCode();
256 		return ajaxQueue.statusCode;
257 	},
258 
259 	/**
260 	 * Checks the status code by counting the number of tasks, and checking the value in sendingStatusCode
261 	 * 
262 	 * @private
263 	 * @example
264 	 * ajaxQueue.checkStatusCode()
265 	 */
266 	checkStatusCode: function () {
267 		var i = 0,
268 			oldStatusCode = ajaxQueue.statusCode,
269 			callback;
270 		if(ajaxQueue.getQueueLength() > 0) {
271 			if(ajaxQueue.sendingStatusCode) {
272 				// Is sending, and there is tasks in the queue
273 				ajaxQueue.setStatusCode(2);
274 			} else {
275 				// Is not sending, but there is tasks in the queue
276 				ajaxQueue.setStatusCode(1);
277 			}
278 		} else {
279 			// No tasks in queue, and is therefore not sending
280 			ajaxQueue.setStatusCode(0);
281 		}
282 
283 		if(oldStatusCode !== ajaxQueue.statusCode) {
284 			if(ajaxQueue.callbackArray.onStatusCodeChange.length > 0) {
285 				for (i = 0; i <= ajaxQueue.callbackArray.onStatusCodeChange.length - 1; i++) {
286 					callback = ajaxQueue.callbackArray.onStatusCodeChange[i];
287 					if (callback && typeof (callback) === "function") {
288 						callback();
289 					}
290 				};
291 			}
292 		}
293 	},
294 
295 	/**
296 	 * Executes the tasks in the queue
297 	 * 
298 	 * @static
299 	 * @example
300 	 * ajaxQueue.executeTasks();
301 	 */
302 	executeTasks: function () {
303 		if (this.queueArray.Tasks.length > 0) {
304 			ajaxQueue.sendingStatusCode = true;
305 			ajaxQueue.checkStatusCode();
306 			var currentTask = this.queueArray.Tasks[0],
307 				callback = null,
308 				result = {};
309 			ajaxQueue.log("Sending '" + currentTask.data + "' to '" + currentTask.url + "'.");/*LOG*/
310 			$.ajax({
311 				type: 'POST',
312 				url: currentTask.url,
313 				data: currentTask.data,
314 				timeout: ajaxQueue.ajaxTimeout,
315 				// On success
316 				success: function (data) {
317 					// Remove the task from the queue
318 					ajaxQueue.remove(currentTask.id);
319 					ajaxQueue.log("Sending of '" + currentTask.data + "' to '" + currentTask.url + "' was successfull.");/*LOG*/
320 					// Call the groups callback
321 					callback = ajaxQueue.callbackArray.onSuccess[currentTask.group];
322 					if (callback && typeof (callback) === "function") {
323 						ajaxQueue.log("Calling back!");
324 						result.data = data;
325 						result.url = currentTask.url;
326 						result.sentdata = currentTask.data;
327 						callback(result);
328 					}
329 					// Loop on...
330 					ajaxQueue.executeTasks();
331 				},
332 				error: function (jqXHR, textStatus, errorThrown) {
333 					// Timeout
334 					if (textStatus === "timeout") {
335 						callback = ajaxQueue.callbackArray.onTimeout[currentTask.group];
336 						if (callback && typeof (callback) === "function") {
337 							ajaxQueue.log("Calling back!");
338 							callback();
339 						}
340 					}
341 					if (textStatus === "error" || textStatus === "abort" || textStatus === "parsererror") {
342 						callback = ajaxQueue.callbackArray.onError[currentTask.group];
343 						if (callback && typeof (callback) === "function") {
344 							ajaxQueue.log("Calling back!");
345 							callback();
346 						}
347 					}
348 					if (ajaxQueue.retryTimeout !== false) {
349 						ajaxQueue.log("Coming back around(in a bit)!!!");
350 						// Try againg in a bit
351 						setTimeout("ajaxQueue.executeTasks()", ajaxQueue.retryTimeout);
352 					}
353 				}
354 			});
355 		} else {
356 			ajaxQueue.sendingStatusCode = false;
357 			ajaxQueue.checkStatusCode();
358 		}
359 		return true;
360 	},

363 	 * Set the configuration
364 	 * 
365 	 * @static
366 	 * @param {JSON} config The settings to change
367 	 * @example
368 	 * ajaxQueue.setConfig({
369 	 *  idLength: 5,
370 	 *  localStorageKeyName: "ajaxQueue2",
371 	 *  retryTimeout: false, // Or for example 2000
372 	 *  ajaxTimeout: 5000
373 	 * });
374 	 */
375 	setConfig: function (config) {
376 		// Check for the idLength configuration
377 		if (config.idLength !== undefined) {
378 			ajaxQueue.idLength = config.idLength;
379 		}
380 		// Check for the localStorageKeyName configuration
381 		if (config.localStorageKeyName !== undefined) {
382 			ajaxQueue.localStorageKeyName = config.localStorageKeyName;
383 		}
384 		// Check for the retryTimeout configuration
385 		if (config.retryTimeout !== undefined) {
386 			ajaxQueue.retryTimeout = config.retryTimeout;
387 		}
388 		// Check for the ajaxTimeout configuration
389 		if (config.ajaxTimeout !== undefined) {
390 			ajaxQueue.ajaxTimeout = config.ajaxTimeout;
391 		}

394 	/**
395 	 * Registers a callback function.
396 	 * Callback types:
397 	 * 	onSuccess
398 	 * 	onError
399 	 * 	onTimeout
400 	 * 	onStatusCodeChange
401 	 * 
402 	 * @static
403 	 * @param  {JSON}     options	The json string containing the type and group. Note that the group should not be speecified when using the onStatusCodeChange type.
404 	 * @param  {Function} callback	The function to call
405 	 * @return {Boolean} Whenever the callback could be registered of not
406 	 * @example
407 	 * ajaxQueue.registerCallback({
408 	 *  type: "onSuccess",
409 	 *  group: "testGroup23"
410 	 * }, function () {
411 	 *  alert('Message was delivered successfully!');
412 	 * });
413 	 * @example
414 	 * ajaxQueue.registerCallback({
415 	 *  type: "onSuccess",
416 	 *  group: "testGroup23"
417 	 * }, testFunction);
418 	 * @example
419 	 * ajaxQueue.registerCallback({
420 	 * 	type: "onStatusCodeChange"
421 	 * }, testFunction)
422 	 */
423 	registerCallback: function (json, callback) {
424 		if(json.type === "onStatusCodeChange") {
425 			var current = ajaxQueue.callbackArray[json.type];
426 			current.push(callback);
427 			return true;
428 		}
429 		if (callback && typeof (callback) === "function" && json.type && json.group) {
430 			var current = ajaxQueue.callbackArray[json.type];
431 			current[json.group] = callback;
432 			return true;
433 		} else {
434 			return false;
435 		}
436 	},

441 	 * @private
442 	 * @return {String} The newly created Id

445 	 */
446 	generateId: function () {

450 	/**
451 	 * Generates a random string at the specified length
452 	 *
453 	 * @private
454 	 * @param {Number} length The length of the string
455 	 * @returns {String} The random string
456 	 * @example
457 	 * ajaxQueue.randomString(5); // Returns random string
458 	 */
459 	randomString: function (length) {
460 		var i = 0,
461 			// Define the chars used
462 			chars = "0123456789ABCDEFGHIJKLMNOPQRSTUVWXTZabcdefghiklmnopqrstuvwxyz",
463 			// Initialize variables
464 			randomString = '',
465 			randomNumber = 0;
466 		// Loop *length* times
467 		for (i = 0; i < length; i++) {
468 			// Generate a random number
469 			randomNumber = Math.floor(Math.random() * chars.length);
470 			// Add a char to the random string
471 			randomString += chars.substring(randomNumber, randomNumber + 1);
472 		}
473 		return randomString;
474 	}
