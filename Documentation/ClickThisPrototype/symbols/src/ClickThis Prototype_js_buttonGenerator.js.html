 10 /**
 11 * buttonGenerator Class
 12 * @class buttonGenerator Class
 13 */
 14 "use strict";
 15 var buttonGenerator = {
 16 	/**
 17 	* The default color when the button is on (String)
 18 	*/
 19 	defaultColor: "blue",
 20 	/**
 21 	* Creates a new button
 22 	*
 23 	* @param {integer} id The id of the button
 24 	* @param {integer} value The on/off value (1/0)
 25 	* @param {string} color The color of the button
 26 	* @param {string} text The button text
 27 	* @param {boolean} submit Wherever it should submit its data
 28 	* @param {boolean} single Wherever it should deselect all other buttons (Single choice)
 29 	* @param {string} group The submit group of the button, this parameter is optional
 30 	* @returns {string} The html for the button
 31 	*/
 32 	newButton: function (id, value, color, text, submit, single, group) {
 33 		if (!submit) {
 34 			submit = true;
 35 		}
 36 		if (!single) {
 37 			single = false;
 38 		}
 39 		if (group === undefined) {
 40 			group = null;
 41 		}
 42 		return this.newCustomButton(id, value, color, this.defaultColor, text, text, submit, single, group);
 43 	},
 44 	/**
 45 	* Creates a new custom button
 46 	*
 47 	* @param {integer} id The id of the button
 48 	* @param {integer} value The on/off value (1/0)
 49 	* @param {string} colorOff The color of the button when its off
 50 	* @param {string} colorOn The color of the buttonwhen its on
 51 	* @param {string} textOff The button text when its on
 52 	* @param {string} textOn The button text when its off
 53 	* @param {boolean} submit Wherever it should submit its data
 54 	* @param {boolean} single Wherever it should deselect all other buttons (Single choice)
 55 	* @param {string} group The submit group of the button
 56 	* @returns {string} The html for the button
 57 	*/
 58 	newCustomButton: function (id, value, colorOff, colorOn, textOff, textOn, submit, single, group) {
 59 		var cssClass = "",
 60 			groupHTML = "",
 61 			currentColor = "",
 62 			currentText = "",
 63 			onClickFunctions = "",
 64 			specialClass = "",
 65 			html = "";
 66 
 67 		// Get the current color for the button
 68 		if (value) {
 69 			currentColor = colorOn;
 70 		} else {
 71 			currentColor = colorOff;
 72 		}
 73 		// Get the current text for the button
 74 		if (value) {
 75 			currentText = textOn;
 76 		} else {
 77 			currentText = textOff;
 78 		}
 79 		// Get the cssClass
 80 		cssClass += "mega button " + currentColor + " halfsize ";
 81 		// If it is a submittable button, add submit Class
 82 		if (submit && submit !== undefined && submit !== null) {
 83 			cssClass += "submit ";
 84 		}
 85 		// If it is a single-selectable button, add single Class
 86 		if (single && single !== undefined && single !== null) {
 87 			cssClass += "single ";
 88 		}
 89 		cssClass = $.trim(cssClass);
 90 		// Get the javascript functions
 91 		if (single && single !== undefined && single !== null) {
 92 			onClickFunctions += "buttonGenerator.singleChoice(this);";
 93 		} else {
 94 			onClickFunctions += "buttonGenerator.multipleChoice(this);";
 95 		}
 96 		// Special Classes
 97 		if (single && single !== undefined && single !== null) {
 98 			specialClass = "data-specialClass=\"single\"";
 99 		}
100 		if (group !== undefined && group !== null && group !== "") {
101 			groupHTML = 'data-submitgroup="' + group + '"';
102 		}
103 		// Create Html Code
104 		html = [
105 			'<a class="' + cssClass + '"',
106 			'onClick="' + onClickFunctions + '"',
107 			'data-value="' + value + '"',
108 			'data-id="' + id + '"',
109 			groupHTML,
110 			'data-coloroff="' + colorOff + '"',
111 			'data-coloron="' + colorOn + '"',
112 			'data-textoff="' + textOff + '"',
113 			'data-texton="' + textOn + '"',
114 			specialClass,
115 			'>' + currentText + '</a>\r\n'
116 		].join("");
117 		// Return the Html Code
118 		return html;
119 	},
120 	/**
121 	* Creates a new submit button
122 	*
123 	* @param {integer} id The color of the button
124 	* @param {integer} value The button text
125 	* @returns {string} The html for the button
126 	*/
127 	newSubmitButton: function (color, text) {
128 		var html = [
129 				'<a  class="mega button ' + color + ' halfsize fullsize"',
130 				'onClick="buttonGenerator.submitData();"',
131 				'>' + text + '</a>'
132 			].join("");
133 		// Return the Html Code
134 		return html;
135 	},
136 	/**
137 	* Creates a submit button, with custom properties
138 	*
139 	* @param {string} color The color of the button
140 	* @param {string} text The text of the button
141 	* @param {string} id The HTML id of the button
142 	* @param {string} location The location to submit the result
143 	* @param {string} href The url to redirect, when data is submitted
144 	* @param {string} group The submit group of the button
145 	* @returns {string} The html of the button
146 	*/
147 	newCustomSubmitButton: function (color, text, id, location, href, group) {
148 		var html = [
149 				'<a  class="mega button ' + color + ' halfsize fullsize"',
150 				'onClick="buttonGenerator.submitCustomData(this);"'
151 			].join("");
152 		if (id !== undefined && id !== null) {
153 			html += 'id="' + id + '"';
154 		}
155 		if (location !== undefined && location !== null) {
156 			html += 'data-location="' + location + '"';
157 		}
158 		if (href !== undefined && href !== null) {
159 			html += 'data-link="' + href + '"';
160 		}
161 		if (group !== undefined && group !== null) {
162 			html += 'data-submitgroup="' + group + '"';
163 		}
164 		html += '\\>' + text + '</a>';
165 		// Return the Html Code
166 		return html;
167 	},
168 	/**
169 	* Changes the button's state, from On to Off or from Off to On
170 	*
171 	* @param {object} button The button that it should change state on
172 	*/
173 	changeState: function (button) {
174 		var i = 0,
175 			value = button.getAttribute("data-value"),
176 			colorOff = button.getAttribute("data-coloroff"),
177 			colorOn = button.getAttribute("data-coloron"),
178 			textOff = button.getAttribute("data-textoff"),
179 			textOn = button.getAttribute("data-texton"),
180 			specialClass = button.getAttribute("data-specialclass"),
181 			singleButton = null;
182 
183 		if (specialClass) {
184 			specialClass = " " + specialClass;
185 		} else {
186 			specialClass = "";
187 		}
188 		if (value === "1") {
189 			button.className = 'mega button ' + colorOff + ' halfsize submit' + specialClass;
190 			button.setAttribute("data-value", "0");
191 			button.innerHTML = textOff;
192 		} else {
193 			button.className = 'mega button ' + colorOn + ' halfsize submit' + specialClass;
194 			button.setAttribute("data-value", "1");
195 			button.innerHTML = textOn;
196 		}
197 		if (specialClass !== " single") {
198 			for (i in $('.single').toArray()) {
199 				singleButton = $('.single').toArray()[i];
200 				if (singleButton !== null) {
201 					if (singleButton.getAttribute("data-value") === "1") {
202 						this.changeState(singleButton);
203 					}
204 				}
205 			}
206 		}
207 	},
208 	/**
209 	* Submits the data
210 	*/
211 	submitData: function () {
212 		var i = 0,
213 			postString = "",
214 			button = null;
215 		for (i in $('.button.submit').toArray()) {
216 			button = $('.button.submit').toArray()[i];
217 			if (button !== null) {
218 				postString += button.getAttribute("data-id") + "=" + button.getAttribute("data-value") + ", ";
219 			}
220 		}
221 		postString = postString.slice(0, -1);
222 	},
223 	/**
224 	* Submit the button data with a posibility to use a defined redirect url and a submit location.
225 	* This function also have the ability only to submit data from buttons within the same submit group as the submitButton.
226 	* Mind this function can have some problems and need revalidation and testing.
227 	*
228 	* @param {object} submitButton The submit button that submits
229 	*/
230 	submitCustomData: function (submitButton) {
231 		var i = 0,
232 			postString = "",
233 			postLocation = submitButton.getAttribute("data-location"),
234 			redirectUrl = submitButton.getAttribute("data-link"),
235 			submitGroup = submitButton.getAttribute("data-submitgroup"),
236 			button = null;
237 		for (i in $('.button.submit').toArray()) {
238 			button = $('.button.submit').toArray()[i];
239 			if (submitGroup !== undefined) {
240 				if (button !== null && button.getAttribute("data-submitgroup") === submitGroup) {
241 					if (button.getAttribute("data-id") !== null && button.getAttribute("data-id") !== 'null') {
242 						postString += button.getAttribute("data-id") + "=" + button.getAttribute("data-value") + ", ";
243 					}
244 				}
245 			} else {
246 				if (button !== null && button.getAttribute("data-id") !== null) {
247 					postString += button.getAttribute("data-id") + "=" + button.getAttribute("data-value") + ", ";
248 				}
249 			}
250 		}
251 		postString = postString.slice(0, -1);
252 		if (redirectUrl !== undefined && redirectUrl !== null) {
253 			window.location = redirectUrl;
254 		}
255 	},
256 	/**
257 	* Uncheck all buttons
258 	*/
259 	unCheckAll: function () {
260 		var i = 0,
261 			button = null;
262 		for (i in $('.button.submit').toArray()) {
263 			button = $('.button.submit').toArray()[i];
264 			if (button !== null) {
265 				if (button.getAttribute("data-value") === "1") {
266 					this.changeState(button);
267 				}
268 			}
269 		}
270 	},
271 	/**
272 	* Single Choice button click
273 	*
274 	* @param {object} button The button to perform actions on
275 	*/
276 	singleChoice: function (button) {
277 		var value = button.getAttribute("data-value");
278 		if (value === "1") {
279 			this.unCheckAll();
280 		} else {
281 			this.unCheckAll();
282 			this.changeState(button);
283 		}
284 	},
285 	/**
286 	* Multiple Choice button click
287 	*
288 	* @param {object} button The button to perform actions on
289 	*/
290 	multipleChoice: function (button) {
291 		this.changeState(button);
292 	}
293 };
