 15 var buttonGenerator = {
 16 	/**
 17 	 * The default color when the button is on (String)

 19 	defaultColor: "blue",
 20 	/**
 21 	 * Creates a new button
 22 	 *
 23 	 * @param {integer} id The id of the button
 24 	 * @param {integer} value The on/off value (1/0)
 25 	 * @param {string} color The color of the button
 26 	 * @param {string} text The button text
 27 	 * @param {boolean} submit Wherever it should submit its data
 28 	 * @param {boolean} single Wherever it should deselect all other buttons (Single choice)
 29 	 * @param {string} group The submit group of the button, this parameter is optional
 30 	 * @param {boolean} form If is set to true then it will be a textfield
 31 	 * @param {string} placeholder The form field placeholder text
 32 	 * @param {boolean} spellcheck A boolean to set if the form element spellcheck should be on
 33 	 * @returns {string} The html for the button
 34 	 */
 35 	newButton: function (id, value, color, text, submit, single, group, form, placeholder, spellcheck) {
 36 		if (!submit) {
 37 			submit = true;
 38 		}
 39 		if (!single) {
 40 			single = false;
 41 		}
 42 		if (group === undefined) {
 43 			group = null;
 44 		}
 45 		return this.newCustomButton(id, value, color, text, submit, single, group, form, placeholder, spellcheck);
 46 	},
 47 	/**
 48 	 * Creates a new custom button
 49 	 *
 50 	 * @param {integer} id The id of the button
 51 	 * @param {integer} value The on/off value (1/0)
 52 	 * @param {string} color The color of the button when its on
 53 	 * @param {string} text The button text
 54 	 * @param {boolean} submit Wherever it should submit its data
 55 	 * @param {boolean} single Wherever it should deselect all other buttons (Single choice)
 56 	 * @param {string} group The submit group of the button
 57 	 * @param {boolean} form If is set to true then it will be a textfield
 58 	 * @param {string} placeholder The form field placeholder text
 59 	 * @param {boolean} spellcheck A boolean to set if the form element spellcheck should be on
 60 	 * @returns {string} The html for the button
 61 	 */
 62 	newCustomButton: function (id, value, color, text, submit, single, group, form, placeholder, spellcheck) {
 63 		var cssClass = "",
 64 			groupHTML = "",
 65 			currentColor = "",
 66 			currentText = "",
 67 			onClickFunctions = "",
 68 			specialClass = "",
 69 			html = "";
 70 
 71 		spellcheck = spellcheck || false;
 72 		form = form || false;
 73 		placeholder = placeholder || "";
 74 		// Get the current color for the button
 75 		if (value) {
 76 			currentColor = buttonGenerator.defaultColor;
 77 		} else {
 78 			currentColor = color;
 79 		}
 80 		if (form) {
 81 			currentText = '<textarea placeholder="' + placeholder + '" class="textfield" spellcheck="' + spellcheck + '" lang="en" data-value="0" data-id="1" data-submitgroup="1"></textarea>';
 82 		} else {
 83 			currentText = text;
 84 		}
 85 
 86 		// Get the cssClass
 87 		cssClass += "mega button hyphenate color-" + currentColor + " halfsize ";
 88 		// If it is a submittable button, add submit Class
 89 		if (submit && submit !== undefined && submit !== null) {
 90 			cssClass += "submit ";
 91 		}
 92 		// If it is a single-selectable button, add single Class
 93 		if (single && single !== undefined && single !== null) {
 94 			cssClass += "single ";
 95 		}
 96 		cssClass = $.trim(cssClass);
 97 		// Get the javascript functions
 98 		if (single && single !== undefined && single !== null) {
 99 			onClickFunctions += "buttonGenerator.singleChoice(this," + form + ");";
100 		} else {
101 			onClickFunctions += "buttonGenerator.multipleChoice(this," + form + ");";
102 		}
103 		// Special Classes
104 		if (single && single !== undefined && single !== null) {
105 			specialClass = "data-specialClass=\"single\"";
106 		}
107 		if (group !== undefined && group !== null && group !== "") {
108 			groupHTML = 'data-submitgroup="' + group + '"';
109 		}
110 		// Create Html Code
111 		html = [
112 			'<a class="' + cssClass + '"',
113 			'onClick="' + onClickFunctions + '"',
114 			'data-value="' + value + '"',
115 			'data-id="' + id + '"',
116 			groupHTML,
117 			'data-color="' + color + '"',
118 			'data-text="' + text + '"',
119 			'lang="en"',
120 			specialClass,
121 			'>' + currentText + '</a>\r\n'
122 		].join("");
123 		// Return the Html Code
124 		return html;
125 	},

136 				'onClick="buttonGenerator.submitData();"',
137 				'>' + text + '</a>'
138 			].join("");
139 		// Return the Html Code

141 	},
142 	/**
143 	 * Creates a submit button, with custom properties
144 	 *
145 	 * @param {string} color The color of the button
146 	 * @param {string} text The text of the button
147 	 * @param {string} id The HTML id of the button
148 	 * @param {string} location The location to submit the result

173 	},
174 	/**
175 	 * This function creates a Submit button that can submit,
176 	 * but doesn't redicret.
177 	 * @param  {[string} color   The wished color of the button
178 	 * @param  {string} text     The text of the button
179 	 * @param  {string} id       The id of the button if wished
180 	 * @param  {string} location The post location of the submit group
181 	 * @param  {string} group    The submit group
182 	 * @param  {string} clickCallbackString The name of the function to call when clicked
183 	 * @param {string} callbackParameters The parameters to the callback function as string
184 	 */
185 	newCustomSwipeSubmitButton : function (color, text, id, location, group, clickCallbackString, callbackParameters) {
186 		var html = [
187 				'<a  class="mega button color-' + color + ' halfsize fullsize"',
188 				'onClick="buttonGenerator.submitCustomSwipeData(this);"'
189 			].join("");
190 		if (id !== undefined && id !== null) {
191 			html += 'id="' + id + '"';
192 		}
193 		if (location !== undefined && location !== null) {
194 			html += 'data-location="' + location + '"';
195 		}
196 		if (group !== undefined && group !== null) {
197 			html += 'data-submitgroup="' + group + '"';
198 		}
199 		if (clickCallbackString !== undefined && clickCallbackString !== null && typeof clickCallbackString === 'string') {
200 			html += 'data-clickcallback="' + clickCallbackString + '"';

202 		if (callbackParameters !== undefined && callbackParameters !== null && typeof callbackParameters === 'string') {
203 			html += 'data-callbackParameters="' + callbackParameters + '"';

205 		html += '\\>' + text + '</a>';
206 		// Return the Html Code

208 	},
209 	/**
210 	 * Changes the button's state from On to Off, or from Off to On.
211 	 *
212 	 * @param {object} button The button that it should change state on
213 	 * @param {boolean} form Boolean to set if the current element is a form element
214 	 */
215 	changeState: function (button, form) {
216 		// Variable decleration
217 		var i = 0,
218 			value = button.getAttribute("data-value") || "",
219 			color = button.getAttribute("data-color") || "",
220 			text = button.getAttribute("data-text") || "",
221 			specialClass = button.getAttribute("data-specialclass") || "",
222 			singleButton = null,
223 			singleButtons = null,
224 			isFormElement = form || false,
225 			classArray = null;
226 
227 		// Get and convert the elements class string to an array
228 		classArray = $(button).attr("class").split(" ");
229 
230 		// Remove all classes with "color-" in its name
231 		for (i = classArray.length - 1; i >= 0; i--) {
232 			if (classArray[i].indexOf("color-") !== -1) {
233 				classArray.splice(i, 1);
234 			}
235 		}
236 
237 		// Change the state of the button
238 		if (value === "1") {
239 			button.setAttribute("data-value", "0");
240 			classArray.push("color-" + color);
241 			if (!isFormElement) {
242 				//button.innerHTML = text;
243 			}
244 		} else {
245 			button.setAttribute("data-value", "1");
246 			classArray.push("color-" + buttonGenerator.defaultColor);
247 			if (!isFormElement) {
248 				//button.innerHTML = text;
249 			}
250 		}
251 
252 		// Join the class array, and put it back on the element
253 		$(button).attr("class", classArray.join(" "));
254 
255 		// Create an array of single choice buttons
256 		singleButtons = $('.single').toArray();
257 
258 		// If this is a single choice button 
259 		if (specialClass.indexOf("single") === -1) {
260 			for (i in singleButtons) {
261 				singleButton = singleButtons[i];
262 				if (singleButton !== null) {
263 					if (singleButton.getAttribute("data-value") === "1") {
264 						this.changeState(singleButton);
265 					}
266 				}
267 			}
268 		}
269 	},
270 	/**
271 	 * Submits the data
272 	 */
273 	submitData: function () {
274 		var i = 0,
275 			postString = "",
276 			button = null;
277 		for (i in $('.button.submit').toArray()) {
278 			button = $('.button.submit').toArray()[i];
279 			if (button !== null) {
280 				postString += button.getAttribute("data-id") + "=" + button.getAttribute("data-value") + ", ";
281 			}
282 		}
283 		postString = postString.slice(0, -1);
284 	},
285 	/**
286 	 * Submit the button data with a posibility to use a defined redirect url and a submit location.
287 	 * This function also have the ability only to submit data from buttons within the same submit group as the submitButton.
288 	 * Mind this function can have some problems and need revalidation and testing.
289 	 *
290 	 * @param {object} submitButton The submit button that submits
291 	 */
292 	submitCustomData: function (submitButton) {
293 		var i = 0,
294 			postString = "",
295 			postLocation = submitButton.getAttribute("data-location"),
296 			redirectUrl = submitButton.getAttribute("data-link"),
297 			submitGroup = submitButton.getAttribute("data-submitgroup"),
298 			button = null;
299 		for (i in $('.button.submit').toArray()) {
300 			button = $('.button.submit').toArray()[i];
301 			if (submitGroup !== undefined) {
302 				if (button !== null && button.getAttribute("data-submitgroup") === submitGroup) {
303 					if (button.getAttribute("data-id") !== null && button.getAttribute("data-id") !== 'null') {
304 						postString += button.getAttribute("data-id") + "=" + button.getAttribute("data-value") + ", ";
305 					}
306 				}
307 			} else {
308 				if (button !== null && button.getAttribute("data-id") !== null) {
309 					postString += button.getAttribute("data-id") + "=" + button.getAttribute("data-value") + ", ";
310 				}
311 			}
312 		}
313 		postString = postString.slice(0, -1);
314 		if (redirectUrl !== undefined && redirectUrl !== null) {
315 			window.location = redirectUrl;
316 		}
317 	},
318 	/**
319 	 * This function cretes a custom submit button,
320 	 * the only different from 'submitCutstomData' is,
321 	 * that this function doesn't redirect it call a callback function instead
322 	 * @see submitCustomData
323 	 * @param  {object} submitButton The submit button object that is clicked
324 	 */
325 	submitCustomSwipeData : function (submitButton) {
326 		var i = 0,
327 			postString = "",
328 			postLocation = submitButton.getAttribute("data-location"),
329 			callback = submitButton.getAttribute('data-clickcallback'),
330 			submitGroup = submitButton.getAttribute("data-submitgroup"),
331 			callbackOptions = submitButton.getAttribute('data-callbackParameters'),
332 			button = null;
333 		for (i in $('.button.submit').toArray()) {
334 			button = $('.button.submit').toArray()[i];
335 			if (submitGroup !== undefined) {
336 				if (button !== null && button.getAttribute("data-submitgroup") === submitGroup) {
337 					if (button.getAttribute("data-id") !== null && button.getAttribute("data-id") !== 'null') {
338 						postString += button.getAttribute("data-id") + "=" + button.getAttribute("data-value") + ", ";
339 					}
340 				}
341 			} else {
342 				if (button !== null && button.getAttribute("data-id") !== null) {
343 					postString += button.getAttribute("data-id") + "=" + button.getAttribute("data-value") + ", ";
344 				}
345 			}
346 		}
347 
348 		postString = postString.slice(0, -1);
349 		if (callback !== null && callback !== undefined && typeof callback === 'string') {
350 			if (callbackOptions !== undefined && callbackOptions !== null && typeof callbackOptions === 'string') {
351 				eval(callback + '(' + callbackOptions + ')');
352 			} else {
353 				eval(callback + '()');
354 			}
355 		}
356 	},
357 	/**
358 	 * Uncheck all buttons
359 	 */
360 	unCheckAll: function () {
361 		var i = 0,
362 			button = null;
363 		for (i in $('.button.submit').toArray()) {
364 			button = $('.button.submit').toArray()[i];
365 			if (button !== null) {
366 				if (button.getAttribute("data-value") === "1") {
367 					if ($(button).find(".textfield").length > 0) {
368 						this.changeState(button, true);
369 					} else {
370 						this.changeState(button);
371 					}
372 				}
373 			}
374 		}
375 	},
376 	/**
377 	 * Single Choice button click
378 	 *
379 	 * @param {object} button The button to perform actions on
380 	 * @param {boolean} form A boolean to set if the element is a form element
381 	 */
382 	singleChoice: function (button, form) {
383 		form = form || false;
384 		var value = button.getAttribute("data-value");
385 		if (value === "1") {
386 			this.unCheckAll();
387 		} else {
388 			this.unCheckAll();
389 			this.changeState(button, form);
390 		}
391 	},
392 	/**
393 	 * Multiple Choice button click
394 	 *
395 	 * @param {object} button The button to perform actions on
396 	 * @param {boolean} form A boolean to set if the element is a form element
397 	 */
398 	multipleChoice: function (button, form) {
399 		form = form || false;
400 		this.changeState(button, form);
401 	}
