 14 "use strict";
 15 var buttonGenerator = {
 16 	/**
 17 	 * The default color when the button is on (String)
 18 	 */
 19 	defaultColor: "blue",
 20 	/**
 21 	 * Creates a new button
 22 	 *
 23 	 * @param {integer} id The id of the button
 24 	 * @param {string} color The color of the button
 25 	 * @param {string} text The button text
 26 	 * @param {integer} type The button type
 27 	 * @param {integer} group The button's group
 28 	 * @returns {string} The html for the button
 29 	 */
 30 	newButton: function (id, color, text, type, group) {
 31 		var cssClass = "",
 32 			groupHTML = "",
 33 			currentText = "",
 34 			onClickFunctions = "",
 35 			specialClass = "",
 36 			html = "",
 37 			onClickType = "onclick",
 38 			specialFunctions = "",
 39 			textField = 0;
 40 
 41 		if (type === 3 || type === 4) {
 42 			textField = 1
 43 		}
 44 
 45 		if (type === 3 || type === 4) {
 46 			currentText = '<textarea placeholder="' + text + '" class="textfield" spellcheck="0" lang="en" data-value="0" data-id="1" data-submitgroup="1"></textarea>';
 47 		} else {
 48 			currentText = text;
 49 		}
 50 
 51 		// Get the cssClass
 52 		cssClass += "mega button color-" + color + " halfsize ";
 53 
 54 		// If it is a single-selectable button, add single Class
 55 		if (type === 2) {
 56 			cssClass += "single ";
 57 		}
 58 		// Get the javascript functions
 59 		if (type === 1) {
 60 			onClickFunctions += "buttonGenerator.multipleChoice(this," + textField + ");";
 61 		}
 62 		if (type === 2) { // Single
 63 			onClickFunctions += "buttonGenerator.singleChoice(this," + textField + ");";
 64 		}
 65 		if(type === 3){ // Multi Textfield
 66 			specialFunctions = 'ondblclick="buttonGenerator.multipleChoice(this,' + text + ',true);"';
 67 			cssClass += "fullsize ";
 68 		}	
 69 		if(type === 4){ // Single Textfield
 70 			specialFunctions = 'ondblclick="buttonGenerator.singleChoice(this,' + text + ',true);"';
 71 			cssClass += "fullsize ";
 72 		}
 73 		// Trim the CSS class
 74 		cssClass = $.trim(cssClass);
 75 		// Special Classes
 76 		if (type === 2) { // Single
 77 			specialClass = "data-specialClass=\"single\"";
 78 		}
 79 		if (group !== undefined && group !== null && group !== "") {
 80 			groupHTML = 'data-submitgroup="' + group + '"';
 81 		}
 82 		// Create Html Code
 83 		html = [
 84 			'<a class="' + cssClass + '"',
 85 			onClickType+'="' + onClickFunctions + '"',
 86 			'data-value="0"',
 87 			'data-id="' + id + '"',
 88 			groupHTML,
 89 			'data-color="' + color + '"',
 90 			'data-text="' + text + '"',
 91 			'lang="en"',
 92 			specialClass,
 93 			specialFunctions,
 94 			'>' + currentText + '</a>\r\n'
 95 		].join("");
 96 		// Return the Html Code
 97 		return html;
 98 	},
 99 	/**
100 	 * This function creates a Submit button that can submit,
101 	 * but doesn't redicret.
102 	 * @param  {[string} color   The wished color of the button
103 	 * @param  {string} text     The text of the button
104 	 * @param  {string} id       The id of the button if wished
105 	 * @param  {string} location The post location of the submit group
106 	 * @param  {string} group    The submit group
107 	 * @param  {string} clickCallbackString The name of the function to call when clicked
108 	 * @param {string} callbackParameters The parameters to the callback function as string
109 	 */
110 	newCustomSwipeSubmitButton : function (color, text, id, location, group, clickCallbackString, callbackParameters) {
111 		var html = [
112 				'<a  class="mega button color-' + color + ' halfsize fullsize"',
113 				'onClick="buttonGenerator.submitCustomSwipeData(this);"'
114 			].join("");
115 		if (id !== undefined && id !== null) {
116 			html += 'id="' + id + '"';
117 		}
118 		if (location !== undefined && location !== null) {
119 			html += 'data-location="' + location + '"';
120 		}
121 		if (group !== undefined && group !== null) {
122 			html += 'data-submitgroup="' + group + '"';
123 		}
124 		if (clickCallbackString !== undefined && clickCallbackString !== null && typeof clickCallbackString === 'string') {
125 			html += 'data-clickcallback="' + clickCallbackString + '"';
126 		}
127 		if (callbackParameters !== undefined && callbackParameters !== null && typeof callbackParameters === 'string') {
128 			html += 'data-callbackParameters="' + callbackParameters + '"';
129 		}
130 		html += '\\>' + text + '</a>';
131 		// Return the Html Code
132 		return html;
133 	},
134 	/**
135 	 * Changes the button's state from On to Off, or from Off to On.
136 	 *
137 	 * @param {object} button The button that it should change state on
138 	 * @param {boolean} form Boolean to set if the current element is a form element
139 	 * @param {boolean} formDeselect If this parameter is set to true other form deselect options will be overruled
140 	 */
141 	changeState: function (button, form , formDeselect) {
142 		// Variable decleration
143 		var i = 0,
144 			value = button.getAttribute("data-value") || "",
145 			color = button.getAttribute("data-color") || "",
146 			text = button.getAttribute("data-text") || "",
147 			specialClass = button.getAttribute("data-specialclass") || "",
148 			singleButton = null,
149 			singleButtons = null,
150 			isFormElement = form || false,
151 			classArray = null;
152 
153 		// Get and convert the elements class string to an array
154 		classArray = $(button).attr("class").split(" ");
155 
156 		// Remove all classes with "color-" in its name
157 		for (i = classArray.length - 1; i >= 0; i--) {
158 			if (classArray[i].indexOf("color-") !== -1) {
159 				classArray.splice(i, 1);
160 			}
161 		}
162 
163 		if(form && $(button).find(".textfield").val() != "" && formDeselect != true){
164 			if(value === "1"){
165 				return;
166 			}
167 		}
168 
169 		// Change the state of the button
170 		if (value === "1") {
171 			button.setAttribute("data-value", "0");
172 			classArray.push("color-" + color);
173 			if (!isFormElement) {
174 				//button.innerHTML = text;
175 			}
176 		} else {
177 			button.setAttribute("data-value", "1");
178 			classArray.push("color-" + buttonGenerator.defaultColor);
179 			if (!isFormElement) {
180 				//button.innerHTML = text;
181 			}
182 		}
183 
184 		// Join the class array, and put it back on the element
185 		$(button).attr("class", classArray.join(" "));
186 
187 		// Create an array of single choice buttons
188 		singleButtons = $('.single').toArray();
189 
190 		// If this is a single choice button 
191 		if (specialClass.indexOf("single") === -1) {
192 			for (i in singleButtons) {
193 				singleButton = singleButtons[i];
194 				if (singleButton !== null) {
195 					if (singleButton.getAttribute("data-value") === "1") {
196 						this.changeState(singleButton);
197 					}
198 				}
199 			}
200 		}
201 	},
202 	/**
203 	 * This function cretes a custom submit button,
204 	 * the only different from 'submitCutstomData' is,
205 	 * that this function doesn't redirect it call a callback function instead
206 	 * @see submitCustomData
207 	 * @param  {object} submitButton The submit button object that is clicked
208 	 */
209 	submitCustomSwipeData : function (submitButton) {
210 		var i = 0,
211 			postString = "",
212 			postLocation = submitButton.getAttribute("data-location"),
213 			callback = submitButton.getAttribute('data-clickcallback'),
214 			submitGroup = submitButton.getAttribute("data-submitgroup"),
215 			callbackOptions = submitButton.getAttribute('data-callbackParameters'),
216 			button = null;
217 		for (i in $('.button').toArray()) {
218 			button = $('.button').toArray()[i];
219 			if (submitGroup !== undefined) {
220 				if (button !== null && button.getAttribute("data-submitgroup") === submitGroup) {
221 					if (button.getAttribute("data-id") !== null && button.getAttribute("data-id") !== 'null') {
222 						postString += button.getAttribute("data-id") + "=" + button.getAttribute("data-value") + ", ";
223 					}
224 				}
225 			} else {
226 				if (button !== null && button.getAttribute("data-id") !== null) {
227 					postString += button.getAttribute("data-id") + "=" + button.getAttribute("data-value") + ", ";
228 				}
229 			}
230 		}
231 
232 		postString = postString.slice(0, -1);
233 		if (callback !== null && callback !== undefined && typeof callback === 'string') {
234 			if (callbackOptions !== undefined && callbackOptions !== null && typeof callbackOptions === 'string') {
235 				eval(callback + '(' + callbackOptions + ')');
236 			} else {
237 				eval(callback + '()');
238 			}
239 		}
240 	},
241 	/**
242 	 * Uncheck all buttons
243 	 */
244 	unCheckAll: function () {
245 		var i = 0,
246 			button = null;
247 		for (i in $('.button').toArray()) {
248 			button = $('.button').toArray()[i];
249 			if (button !== null) {
250 				if (button.getAttribute("data-value") === "1") {
251 					if ($(button).find(".textfield").length > 0) {
252 						this.changeState(button, true);
253 					} else {
254 						this.changeState(button);
255 					}
256 				}
257 			}
258 		}
259 	},
260 	/**
261 	 * Single Choice button click
262 	 *
263 	 * @param {object} button The button to perform actions on
264 	 * @param {boolean} form A boolean to set if the element is a form element
265 	 * @param {boolean} formDeselect If this is set to true all other form select/deselect options will be overwritten
266 	 */
267 	singleChoice: function (button, form, formDeselect) {
268 		form = form || false;
269 		formDeselect = formDeselect || false;
270 		var value = button.getAttribute("data-value");
271 		if (value === "1") {
272 			this.unCheckAll();
273 		} else {
274 			this.unCheckAll();
275 			this.changeState(button, form, formDeselect);
276 		}
277 	},
278 	/**
279 	 * Multiple Choice button click
280 	 *
281 	 * @param {object} button The button to perform actions on
282 	 * @param {boolean} form A boolean to set if the element is a form element
283 	 * @param {boolean} formDeselect If this is set to true all other form select/deselect options will be overwritten
284 	 */
285 	multipleChoice: function (button, form, formDeselect) {
286 		form = form || false;
287 		formDeselect = formDeselect || false;
288 		this.changeState(button, form, formDeselect);
289 	}
290 };
