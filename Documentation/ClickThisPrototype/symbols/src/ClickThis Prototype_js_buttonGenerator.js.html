 10 /**
 11  * buttonGenerator Class
 12  * @class buttonGenerator Class
 13  */
 14 "use strict";
 15 var buttonGenerator = {
 16 	/**
 17 	 * The default color when the button is on (String)
 18 	 */
 19 	defaultColor: "blue",
 20 	/**
 21 	 * Creates a new button
 22 	 *
 23 	 * @example
 24 	 * buttonGenerator.newButton({
 25 	 *  id: 1, // The id of the button
 26 	 *  color: "black", // The color of the button
 27 	 *  text: "Lorem Ipsum", // The text of the button
 28 	 *  type: 2, // The type of the button. 1 = multi, 2 = single, 3 = multi textfield and 4 = single textfield
 29 	 *  group: 1, // The group of the button
 30 	 *  size: 1 // This is the size, 1 = fullsize and 2 = halfsize,
 31 	 *  value: 0 // This is the selection. 0 indicates that the button is not selected, otherwise 1.
 32 	 * })
 33 	 * @returns {string} The html for the button
 34 	 */
 35 	newButton: function (json) {
 36 		var cssClasses = [],
 37 			groupHTML = "",
 38 			currentText = "",
 39 			onClickFunctions = "",
 40 			specialClass = "",
 41 			html = "",
 42 			specialFunctions = "",
 43 			textfield = 0,
 44 			size = json.size || 2;
 45 
 46 		if (json.type === 3 || json.type === 4) {
 47 			textfield = 1;
 48 		}
 49 
 50 		// Make sure that json.value is difined
 51 		if (json.value === undefined) {
 52 			json.value = 0;
 53 		}
 54 
 55 		if (json.type === 3 || json.type === 4) {
 56 			currentText = '<textarea placeholder="' + json.text + '" class="textfield" spellcheck="0" lang="en" data-value="0" data-id="1" data-submitgroup="1"></textarea>';
 57 		} else {
 58 			currentText = json.text;
 59 		}
 60 
 61 		// Declare the basic CSS classes
 62 		cssClasses.push("mega", "button", "color-" + json.color, "size-" + size);
 63 
 64 		// Check type, add the correct class
 65 		if (json.type === 1 || json.type === 3) {
 66 			cssClasses.push("multi");
 67 		} else if (json.type === 2 || json.type === 4) {
 68 			cssClasses.push("single");
 69 		}
 70 
 71 		// Get the javascript functions
 72 		if (json.type === 1) {
 73 			onClickFunctions += "buttonGenerator.multipleChoice(this," + textfield + ");";
 74 		}
 75 		if (json.type === 2) { // Single
 76 			onClickFunctions += "buttonGenerator.singleChoice(this," + textfield + ");";
 77 		}
 78 		if (json.type === 3) { // Multi Textfield
 79 			cssClasses.push("textfield");
 80 		}
 81 		if (json.type === 4) { // Single Textfield
 82 			cssClasses.push("textfield");
 83 		}
 84 		// Special Classes
 85 		if (json.type === 2) { // Single
 86 			specialClass = "data-specialClass=\"single\"";
 87 		}
 88 		// Create Html Code
 89 		html = [
 90 			'<a class="' + cssClasses.join(" ") + '"',
 91 			'onclick="' + onClickFunctions + '"',
 92 			'data-value="' + json.value  + '"',
 93 			'data-id="' + json.id + '"',
 94 			'data-color="' + json.color + '"',
 95 			specialFunctions,
 96 			'>' + currentText + '</a>\r\n'
 97 		].join("");
 98 		// Return the Html Code
 99 		return html;
100 	},
101 	/**
102 	 * Changes the button's state from On to Off, or from Off to On.
103 	 *
104 	 * @param {object} button The button that it should change state on
105 	 * @param {boolean} form Boolean to set if the current element is a form element
106 	 * @param {boolean} formDeselect If this parameter is set to true other form deselect options will be overruled
107 	 */
108 	changeState: function (button, form, formDeselect) {
109 		// Variable decleration
110 		var i = 0,
111 			value = button.getAttribute("data-value") || "",
112 			color = button.getAttribute("data-color") || "",
113 			text = button.getAttribute("data-text") || "",
114 			specialClass = button.getAttribute("data-specialclass") || "",
115 			singleButton = null,
116 			singleButtons = null,
117 			isFormElement = form || false,
118 			classArray = null;
119 
120 		// Get and convert the elements class string to an array
121 		classArray = $(button).attr("class").split(" ");
122 
123 		// Remove all classes with "color-" in its name
124 		for (i = classArray.length - 1; i >= 0; i--) {
125 			if (classArray[i].indexOf("color-") !== -1) {
126 				classArray.splice(i, 1);
127 			}
128 		}
129 
130 		if (form && $(button).find(".textfield").val() !== "" && formDeselect !== true) {
131 			if (value === "1") {
132 				return;
133 			}
134 		}
135 
136 		// Change the state of the button
137 		if (value === "1") {
138 			button.setAttribute("data-value", "0");
139 			classArray.push("color-" + color);
140 		} else {
141 			button.setAttribute("data-value", "1");
142 			classArray.push("color-" + buttonGenerator.defaultColor);
143 		}
144 
145 		// Join the class array, and put it back on the element
146 		$(button).attr("class", classArray.join(" "));
147 
148 		// Create an array of single choice buttons
149 		singleButtons = $('.single').toArray();
150 
151 		// If this is a single choice button 
152 		if (!$(button).hasClass("single")) {
153 			for (i in singleButtons) {
154 				singleButton = singleButtons[i];
155 				if (singleButton !== null) {
156 					if (singleButton.getAttribute("data-value") === "1") {
157 						this.changeState(singleButton);
158 					}
159 				}
160 			}
161 		}
162 	},
163 	/**
164 	 * Uncheck all buttons
165 	 */
166 	unCheckAll: function () {
167 		var i = 0,
168 			button = null;
169 		for (i in $('.button').toArray()) {
170 			button = $('.button').toArray()[i];
171 			if (button !== null) {
172 				if (button.getAttribute("data-value") === "1") {
173 					if ($(button).find(".textfield").length > 0) {
174 						this.changeState(button, true);
175 					} else {
176 						this.changeState(button);
177 					}
178 				}
179 			}
180 		}
181 	},
182 	/**
183 	 * Single Choice button click
184 	 *
185 	 * @param {object} button The button to perform actions on
186 	 * @param {boolean} form A boolean to set if the element is a form element
187 	 * @param {boolean} formDeselect If this is set to true all other form select/deselect options will be overwritten
188 	 */
189 	singleChoice: function (button, form, formDeselect) {
190 		form = form || false;
191 		formDeselect = formDeselect || false;
192 		var value = button.getAttribute("data-value");
193 		if (value === "1") {
194 			this.unCheckAll();
195 		} else {
196 			this.unCheckAll();
197 			this.changeState(button, form, formDeselect);
198 		}
199 	},
200 	/**
201 	 * Multiple Choice button click
202 	 *
203 	 * @param {object} button The button to perform actions on
204 	 * @param {boolean} form A boolean to set if the element is a form element
205 	 * @param {boolean} formDeselect If this is set to true all other form select/deselect options will be overwritten
206 	 */
207 	multipleChoice: function (button, form, formDeselect) {
208 		form = form || false;
209 		formDeselect = formDeselect || false;
210 		this.changeState(button, form, formDeselect);
211 	}
212 };
