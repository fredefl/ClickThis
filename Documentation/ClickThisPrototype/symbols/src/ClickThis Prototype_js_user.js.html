  7 /**
  8  * This function gets the user data from the API, and create's
  9  * the user page if data exists else it show's the not found page.
 10  * @since 1.0
 11  */
 12 function createPage(){
 13 	var userId = getParameterByName("user_id");
 14 	if(typeof userId != "null"){
 15 		$.ajax({
 16 			url: "http://illution.dk/ClickThis/api/user/"+userId,
 17 		  	success: function(data){
 18 		  		if(typeof data.error_code == "undefined"){
 19 		    		$(".Active").addClass("Disabled").removeClass("Active");
 20 		    		var user = $("#user");
 21 		    		$("#linktag").attr("title",data.Name);
 22 		    		$("#profile_image").attr("title",data.Name).attr("alt",data.Name).attr("src",data.ProfileImage);
 23 		    		$("#Name").attr("value",data.Name);
 24 		    		$("#Email").val(data.Email);
 25 		    		$("#Location").val(data.Country);
 26 		    		user.addClass("Active").removeClass("Disabled");
 27 		    	} else {
 28 		    		$(".Active").addClass("Disabled").removeClass("Active");
 29 		    		$("#notfound").addClass("Active").removeClass("Disabled");
 30 		    	}
 31 		  	},
 32 		  	error: function(){
 33 	  			$(".Active").addClass("Disabled").removeClass("Active");
 34 	    		$("#notfound").addClass("Active").removeClass("Disabled");
 35 		  	}
 36 		});
 37 	}
 38 }

 40 /**
 41  * This function returns the value of the "name" uri parameter
 42  * @param  {string} name The url paramerter to get the name of
 43  * @return {string|null}
 44  */
 45 function getParameterByName(name)
 46 {
 47 	name = name.replace(/[\[]/, "\\\[").replace(/[\]]/, "\\\]");
 48 	var regexS = "[\\?&]" + name + "=([^&#]*)";
 49 	var regex = new RegExp(regexS);
 50   	var results = regex.exec(window.location.search);
 51   	if(results == null)
 52     	return null;
 53   	else
 54     	return decodeURIComponent(results[1].replace(/\+/g, " "));
 55 }
