 10 /**
 11  * seriesGenerator Class
 12  * @class seriesGenerator Class
 13  */
 14 "use strict";
 15 
 16 var seriesGenerator = {
 17 	/**
 18 	 * Generates the start page
 19 	 * @param  {string} text The the text that welcomes the user
 20 	 * @return {string}
 21 	 */
 22 	generateStart : function (text) {
 23 		// Append start of div
 24 		var html = '<div id="welcome">';
 25 		// Append the text
 26 		html += '<h1>' + text + '</h1>';
 27 		// Append the begin the survey button
 28 		html += '<a class="mega button color-orange halfsize fullsize" id="begin">Begin the survey</a>';
 29 		// Append end of div
 30 		html += '</div>';
 31 		// Return the markup
 32 		return html;
 33 	},
 34 
 35 	/**
 36 	 * Generates the end page
 37 	 * @param  {string} text The text that says thank you to the user
 38 	 * @return {sting}
 39 	 */
 40 	generateEnd: function (text) {
 41 		// Append start of div
 42 		var html = '<div id="end">';
 43 		// Append the text
 44 		html += '<h1>' + text + '</h1>';
 45 		// Append the end the survey button
 46 		html += '<a class="mega button color-orange halfsize fullsize" id="end_survey">End the survey</a>';
 47 		// Append end of div
 48 		html += '</div>';
 49 		// Return the markup
 50 		return html;
 51 	},
 52 
 53 	/**
 54 	 * Generates a single question
 55 	 * @param  {json} data The question data
 56 	 * @return {string}
 57 	 */
 58 	generateQuestion: function (data) {
 59 		// Append the container div
 60 		var i,
 61 			html = '<div id="question_' + data.Id + '">',
 62 			options = [],
 63 			option = null;
 64 		// Append title
 65 		html += '<h1>' + data.Title + '</h1>';
 66 		// Sort the options
 67 		options = data.Options;
 68 		options.sort(function (a, b) {
 69 			// This sort function sorts so that those items with the lowest view order is first
 70 			if (parseInt(a.ViewOrder, 10) < parseInt(b.ViewOrder, 10)) { // A is first
 71 				return -1;
 72 			} else if (parseInt(a.ViewOrder, 10) > parseInt(b.ViewOrder, 10)) { // B is first
 73 				return 1;
 74 			} else {
 75 				return 0;
 76 			}
 77 		});
 78 		// Append all the options
 79 		for (i = 0; i < options.length; i++) {
 80 			option = options[i];
 81 			html += buttonGenerator.newButton(option.Id, option.Color, option.Title, parseInt(option.OptionType, 10), 1);
 82 		}
 83 
 84 		// End the conainer div
 85 		html += '</div>';
 86 		// Return the markup
 87 		return html;
 88 	},
 89 
 90 	/**
 91 	 * Generates all the quesions
 92 	 * @param  {json} data The questions
 93 	 * @return {string}
 94 	 */
 95 	generateQuestions: function (data) {
 96 		var i,
 97 			html = "";
 98 		// Generates questions, question by question
 99 		for (i = 0; i < data.length; i++) {
100 			html += this.generateQuestion(data[i]);
101 		}
102 		return html;
103 	},
104 
105 	/**
106 	 * Generates html code
107 	 * @param  {json} data The series data
108 	 * @return {string}
109 	 */
110 	generateHtml: function (data) {
111 		// Append container div
112 		var html = "<div>";
113 		// Append the start page
114 		html += this.generateStart(data.StartText);
115 		// Append the questions
116 		html += this.generateQuestions(data.Questions);
117 		// Append the end page
118 		html += this.generateEnd(data.EndText);
119 		// End the container div
120 		html += "</div>";
121 		return html;
122 	},
123 
124 	/**
125 	 * Generates a series
126 	 * @param  {json} data The series data
127 	 * @return {void}
128 	 */
129 	generate: function (data, container) {
130 		// Get the container if not set
131 		if (container === undefined) {
132 			container = $("#questionsContainer");
133 		}
134 		// Add the html
135 		$(container).html(this.generateHtml(data));
136 		// Hyphenate
137 		this.hyphenate(document.body);
138 		// Add swipe functionality
139 		this.addSwipe(container[0]); // The [0] converts the jQuery object to a DOM object
140 		// Add event listeners
141 		this.addListeners();
142 		// Return
143 		return true;
144 	},
145 
146 	/**
147 	 * Runs the hyphenator
148 	 * @param  {object} element The element to hyphenate
149 	 * @return {void}
150 	 */
151 	hyphenate: function (element) {
152 		if (element === undefined) {
153 			element = document.body;
154 		}
155 		// Configurate Hyphenator.js
156 		Hyphenator.config({
157 			displaytogglebox : true,
158 			minwordlength : 4,
159 			useCSS3hyphenation: true,
160 			onhyphenationdonecallback : function () {
161 				buttonResizer.resizeButtons(element);
162 			}
163 		});
164 		// Run the hyphenation
165 		Hyphenator.run();
166 	},
167 
168 	/**
169 	 * Adds swipe functionality
170 	 * @param {object} container The container element to add swipe to
171 	 */
172 	addSwipe: function (element) {
173 		// Initialize the swipe functionality
174 		window.questionSwipe = new Swipe(element, {
175 			callback: function () {
176 				buttonResizer.resizeButtons(document.body);
177 				window.scrollTo(0, 1);
178 			}
179 		});
180 	},
181 
182 	/**
183 	 * Adds button event listeners
184 	 */
185 	addListeners: function () {
186 		// The begin button
187 		$('#begin').click(function () {
188 			window.questionSwipe.next();
189 		});
190 		// The end button
191 		$('#end').click(function () {
192 			window.location = 'http://illution.dk/ClickThisPrototype/home.html';
193 		});
194 	}
195 };
