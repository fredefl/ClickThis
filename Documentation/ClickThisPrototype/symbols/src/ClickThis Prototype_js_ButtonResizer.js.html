 10 /**
 11  * buttonResizer Class
 12  * @class buttonResizer Class
 13  */
 14 "use strict";
 15 var buttonResizer = {
 16 	/**
 17 	 * The initial height of a button when padding should be added.
 18 	 * THIS IS NOT THE STANDARD BUTTON HEIGHT!!!
 19 	 */
 20 	mobileInitialHeigth: 10,
 21 	/**
 22 	 * The level height of a button when padding should be added.
 23 	 * This the size per button resizing.
 24 	 * THIS IS NOT THE STANDARD BUTTON HEIGHT!!!
 25 	 */
 26 	mobileLevelHeigth: 13,
 27 	/**
 28 	 * The initial height of a button when padding should be added.
 29 	 * THIS IS NOT THE STANDARD BUTTON HEIGHT!!!
 30 	 */
 31 
 32 	desktopInitialHeigth: 20,
 33 	/**
 34 	 * The level height of a button when padding should be added. 
 35 	 * This the size per button resizing.
 36 	 * THIS IS NOT THE STANDARD BUTTON HEIGHT!!!
 37 	 */
 38 	desktopLevelHeigth: 32,
 39 
 40 	/**
 41 	 * Resizes the buttons in the specified element.
 42 	 *
 43 	 * @param {element} element The element to search for buttons in
 44 	 * @returns {bool} Wherever it found buttons in the element or not.
 45 	 */
 46 	resizeButtons: function (element) {
 47 		// Initialize arrays & variables
 48 		var i = 0,
 49 			elementArray = new Array(),
 50 			elementArrayLength = 0,
 51 			mobile = false,
 52 			button1 = null,
 53 			button2 = null,
 54 			button1Heigth = 0,
 55 			button2Heigth = 0,
 56 			initialHeigth = 0,
 57 			levelHeigth = 0,
 58 			buttonHeigth = 0,
 59 			biggestButtonHeigth = 0,
 60 			smallestButtonHeigth = 0,
 61 			buttonToResize = null,
 62 			resizeLevel = 0,
 63 			topAddition = 0,
 64 			paddingAddition = 0;
 65 		// Find all buttons in element and stuff them into the array
 66 		$(element).find('.button.mega').each(function (index, element) {
 67 			// Remove padding and top
 68 			$(element).removeAttr('style');
 69 			// Add the element to the array
 70 			elementArray[index] = element;
 71 		});
 72 		// If the is buttons
 73 		if (elementArray.length > 0) {
 74 			// Set loop time/length
 75 			elementArrayLength = elementArray.length;
 76 			// Loop through buttons to find and remove the buttons with no height
 77 			for (i = elementArrayLength - 1; i >= 0; i--) {
 78 				// Define current element
 79 				element = elementArray[i];
 80 				// Get element height
 81 				if ($(element).height() === 0 || $(element).height() === undefined || $(element).height() === null) {
 82 					// Remove element if it has no height
 83 					elementArray.splice(i, 1);
 84 				}
 85 			}
 86 			// Set loop time/length
 87 			elementArrayLength = elementArray.length;
 88 			// Loop though the array and find all the fullsize buttons and remove them
 89 			for (i = 0; i <= elementArrayLength; i++) {
 90 				element = elementArray[i];
 91 				if ($(element).hasClass("fullsize")) {
 92 					if ((i + 1) % 2 === 0) {
 93 						// If there is only one button above.
 94 						elementArray.splice(i - 1, 1);
 95 					}
 96 					elementArray.splice(i, 1);
 97 				}
 98 			}
 99 			// Remove the last button if the total button count is odd/uneven
100 			if (elementArray.length % 2 !== 0) {
101 				elementArray.splice(elementArray.length - 1, 1);
102 			}
103 			// Check if mobile
104 			if ($(elementArray[0]).height() % 26 === 0) {
105 				mobile = true;
106 			}
107 			// Loop through button pairs
108 			for (i = 0; i <= elementArray.length - 1; i += 2) {
109 				// Get buttons
110 				button1 = elementArray[i];
111 				button2 = elementArray[i + 1];
112 				// Get button heigth
113 				button1Heigth = $(button1).height();
114 				button2Heigth = $(button2).height();
115 				// If they have different heigths
116 				if (button1Heigth !== button2Heigth) {
117 					// Set the initial and level height's
118 					if (mobile) {
119 						initialHeigth = this.mobileInitialHeigth;
120 						levelHeigth = this.mobileLevelHeigth;
121 					} else {
122 						initialHeigth = this.desktopInitialHeigth;
123 						levelHeigth = this.desktopLevelHeigth;
124 					}
125 					// Define value
126 					buttonHeigth = levelHeigth * 2;
127 					if (button1Heigth > button2Heigth) {
128 						biggestButtonHeigth = button1Heigth;
129 						smallestButtonHeigth = button2Heigth;
130 						buttonToResize = button2;
131 					} else {
132 						biggestButtonHeigth = button2Heigth;
133 						smallestButtonHeigth = button1Heigth;
134 						buttonToResize = button1;
135 					}
136 					// Calculate the proper height for the button
137 					resizeLevel = ((biggestButtonHeigth - smallestButtonHeigth) / buttonHeigth);
138 					topAddition = resizeLevel * levelHeigth;
139 					paddingAddition = topAddition + initialHeigth;
140 					// Add the extra height(padding) to the button
141 					$(buttonToResize).css("padding", paddingAddition + "px 0 " + paddingAddition + "px 0");
142 					$(buttonToResize).css("top", "-" + topAddition + "px");
143 				}
144 			}
145 		}
146 		// Return if it was a success
147 		if (elementArray.length > 0) {
148 			return true;
149 		} else {
150 			return false;
151 		}
152 	}
153 };
