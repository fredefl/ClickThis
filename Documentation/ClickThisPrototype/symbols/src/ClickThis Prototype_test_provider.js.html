 25  	  * The keyword to put in front of normal pages

 28  	pageKeyword : "page_p", 
 29

 31  	 * The front keyword in front of pages with user content

 34 	userPageKeyword : "user_p",
 35  	
 36  	/**
 37  	 * This function creates a container/ul container and returns it,
 38  	 * if the page object is set then it is append to
 39  	 * @param {object} page  The page object to append to
 40  	 * @param {string} attrclass The wished class to add to the container,
 41  	 * standard value is 'sortable-grid'
 42  	 */
 43  	addContainer : function(page,attrclass) {
 44  		if(typeof attrclass != 'string'){
 45  			attrclass = 'sortable-grid';
 46  		}
 47  		var container = $('<ul></ul>');
 48  		container.addClass(attrclass);
 49  		if(typeof page == 'object'){
 50  			page.append(container);
 51  		}
 52  		return container;
 53 	},
 54 
 55 	/**
 56 	 * This function creates a provider element and an append option is available
 57 	 * @param {object} provider  The JSON object with the provider data
 58 	 * @param {object} container The optional object to append to
 59 	 * @param {string} imageSize The image size
 60 	 * @param {object} data      An optional JSON object with extra data 'element' = 'html attr name'
 61 	 */
 62 	addProvider : function(provider,container,imageSize,data) {
 63 		if (typeof provider == "object") {
 64 			imageSize = imageSize || "128";
 65 			var content = $("<img></img>"); //Makes the content img tag
 66 			var linkTag = $('<a></a>');
 67 			
 68 			//Adds the href to the a tag if its set
 69 			if (provider.Link != undefined && provider.Link != null) {
 70 				linkTag.attr("href",provider.Link);
 71 			}
 72 			
 73 			//Adds the image src if its set
 74 			if (provider.Image != undefined && provider.Image != null) {
 75 				content.attr("src",provider.Image.replace("{size}",imageSize));
 76 			}
 77 			
 78 			//Adds the image title if its set
 79 			if (provider.Title != undefined && provider.Title != null && this.isTouchDevice() == false) {
 80 				var titleTag = $('<span></span>');
 81 				titleTag.append(provider.Title);
 82 				titleTag.addClass("tooltip");
 83 				linkTag.append(titleTag);
 84 			}
 85 
 86 			//Adds the data-provider
 87 			if(provider.Name !== undefined && provider.Name !== null && provider.Name != ''){
 88 				content.attr('data-provider',provider.Name);
 89 			}
 90 			
 91 			//If there is extra attributes to be set, defined in data then set em
 92 			if (typeof data == "object" && data != null) {
 93 				$(data).each(function (index, element) {
 94 	                content.attr(element,provider[element])
 95 	            });
 96 			}
 97 			
 98 			content.addClass('provider');
 99 			linkTag.append(content);
100 			var containerTag = $('<li></li>');
101 			containerTag.append(linkTag);
102 			
103 			//If an append obj is set append the provider to it
104 			if (typeof container == "object" && container != null) {
105 				container.append(containerTag);
106 			}
107 			return containerTag;
108 		}	
109 	},

111 	/**
112 	 * This function creates a 'page' div and return it and if 'object',
113 	 * is an appendable object then it append it to it.
114 	 * @param {object} object The object to append to
115 	 * @param {string} type   The type of the page 'user' or 'default'
116 	 * @param {string} name   The name that will be put after the keyword defined in type
117 	 */
118 	addPage : function(object,type,name) {
119 		var page = $('<div></div>');
120 		if (type == 'default'){
121 			page.addClass('page');
122 			page.addClass('default');
123 			page.attr("id",pageKeyword+name);
124 		} else if(type == "user"){
125 			page.addClass('page');
126 			page.addClass('user');
127 			page.attr("id",userPageKeyword+name);
128 		} else if(type == "show"){
129 			page.addClass('show');
130 			page.attr("id","show_p"+name);
131 		}
132 		if(typeof object == 'object'){
133 			object.append(page);
134 		}
135 		return page;
136 	},

154 		} else if(type == "show"){
155 			page.addClass('show');
156 			page.attr("id","show_p"+name);
157 		}
158 		if(typeof after == 'object'){
159 			after.after(page);
160 		}
161 		return page;

164 	/**
165 	 * This function adds a number of bulletin to a object
166 	 * @param {int} number    the number of bulletins to add
167 	 * @param {int} current   The active page/bulletin
168 	 * @param {object} append    The object to append too
169 	 */
170 	addBullets : function(number,current,append){
171 		if(typeof append == 'object'){
172 			var currentObject;
173 			for (var i = 0; i < number; i++) {
174 				currentObject = $('<em class="bullet">•</em>');
175 				if(i == current){
176 					currentObject.addClass('on');
177 				}
178 				append.append(currentObject);
179 				return currentObject;
180 			};
181 		}

183 	/**
184 	 * This function removes a bullet and if it was the current it find a new current
185 	 * @param  {object} object   The object to remoce
186 	 * @param  {object} container The position container
187 	 */
188 	removeBullet : function(object,container){
189 		if(object.hasClass("on")){
190 			object.remove();
191 			container.find(".bullet:first").addClass("on");
192 		} else {
193 			object.remove();
194 		}
195 	},

204 			append.append(currentObject);
205 			return currentObject;

214 	 */

224 			return newObject;	
225 		}
