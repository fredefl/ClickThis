 15  /**
 16  * This class can add a provider,
 17  * page,container,column,row and a provider tag with image and a tag 
 18  * to a specific jQuery object.
 19   * @class provider Login Provider Class
 20   * @type {Object}
 21   */
 22  var provider = {
 23 
 24 	/**
 25 	 * The keyword to put in front of normal pages
 26 	 * @type {String}
 27 	 */
 28 	pageKeyword : "page_p", 
 29 
 30 	/**
 31 	 * The front keyword in front of pages with user content
 32 	 * @type {String}
 33 	 */
 34 	userPageKeyword : "user_p",
 35 	
 36 	/**
 37 	 * This function creates a table, this function will be used to create the page tables
 38 	 * @param {object} obj This parameter is the jquery object of the outer div or outer container
 39 	 * @since 1.0
 40 	 * @param {object} cellspacing This is an optional parameter, to set the table cellspacing if its not set then the value will be 10
 41 	 * @returns {object} This function returns the jquery object of the container
 42 	 *
 43 	 * @example
 44 	 * //This function uses a jquery object and a optional integer with the cellspacing value
 45 	 * provider.addContainer($('#box'),10)
 46 	 * 
 47 	 * @example
 48 	 * var container = provider.addContainer();
 49 	 * 
 50 	 * @example
 51 	 * //This example will not do anything because, an append object is not defined and nothing is done with the result.
 52 	 * var box = null;
 53 	 * provider.addContainer(box,null);
 54 	 * 
 55 	 * @example
 56 	 * //This example will not append anything but the result will be stored in 'container'.
 57 	 * var container;
 58 	 * container = provider.addProvider(null,10);
 59 	 */
 60 	addContainer : function(obj,cellspacing) {
 61 		var container = $("<table></table>");
 62 		if (typeof cellspacing != "number") {
 63 			cellspacing = 10;
 64 		}
 65 		container.attr("cellspacing",cellspacing);
 66 		if (typeof obj == "object" && obj != null) {
 67 			obj.append(container);
 68 		}
 69 		return container;
 70 	},
 71 
 72 	/**
 73 	 * This function adds the necesary html code for provider element,
 74 	 * and if obj is specified the element will also be append to to obj
 75 	 * @param {object} obj The jquery object you wish to add the provider to
 76 	 * @param {array} data This parameter is optional it's used to specify other rows then standard,
 77 	 * but beaware that the html tag and the class variable must be the same.
 78 	 * @returns {object} The created jquery object of the provider
 79 	 * @since 1.0
 80 	 */
 81 	addProvider : function(provider,obj,data) {
 82 		if (typeof provider == "object") {
 83 			var linkTag = $("<a></a>"); //Makes the a tag
 84 			var content = $("<img></img>"); //Makes the content img tag
 85 			
 86 			//Adds the href to the a tag if its set
 87 			if (provider.Link != undefined && provider.Link != null) {
 88 				linkTag.attr("href",provider.Link);
 89 			}
 90 			
 91 			//Adds the image src if its set
 92 			if (provider.Image != undefined && provider.Image != null) {
 93 				content.attr("src",provider.Image);
 94 			}
 95 			
 96 			//Adds the image alt if its set
 97 			if (provider.Alt != undefined && provider.Alt != null) {
 98 				content.attr("alt",provider.Alt);
 99 			}
100 			
101 			//Adds the image title if its set
102 			if (provider.Title != undefined && provider.Title != null) {
103 				content.attr("title",provider.Title);
104 			}
105 			
106 			//If there is extra attributes to be set, defined in data then set em
107 			if (typeof data == "object" && data != null) {
108 				$(data).each(function (index, element) {
109 					content.attr(element,provider[element])
110 				});
111 			}
112 			
113 			content.addClass('provider');
114 			linkTag.addClass('providerLink');
115 			//Append the img tag to the a tag
116 			linkTag.append(content);
117 			
118 			//If an append obj is set append the provider to it
119 			if (typeof obj == "object" && obj != null) {
120 				obj.append(linkTag);
121 			}
122 			return linkTag;
123 		}	
124 	},
125 
126 	/**
127 	 * This function adds a page to the container with the specified parameters
128 	 * @param {object} obj The container you wish to add the page div to
129 	 * @param {string} type This is the type of the page, "user" or "default"/page
130 	 * @param {string} name The name of the page div the final id will be the choosen keyoed user or page + name
131 	 * @param {string} state The state of the page "Disabled" or "Active"
132 	 * @returns {object} This function returns a jquery object
133 	 * @since 1.0
134 	 */
135 	addPage : function(obj,type,name,state) {
136 		var div = $("<div></div>");
137 		var objectName;
138 		if (type === "default") {
139 			div.addClass('page');
140 			div.addClass('default');
141 			div.attr("id",pageKeyword+name);
142 			objectName = pageKeyword+name;
143 		}
144 		else if (type === "user") {
145 			div.addClass('page');
146 			div.addClass('user');
147 			div.attr("id",userPageKeyword+name);
148 			objectName = userPageKeyword+name;
149 		}
150 		if (typeof obj == "object" && obj != null) {
151 			obj.append(div);
152 		}
153 		return $('#'+objectName);
154 	},
155 
156 	/**
157 	 * This function creates a new row in a table with the tr tags,
158 	 * and returns the jquery object if a table is deffined as an object in obj the row is appended to it.
159 	 * @param {object} The jquery object of the table
160 	 * @returns {object} The jquery object of the new row
161 	 * @since 1.0
162 	 * @example
163 	 * provider.addRow(column)
164 	 */
165 	addRow : function(obj) {
166 		var row = $("<tr></tr>");
167 		if (typeof obj == "object" && obj != null) {
168 			obj.append(row);
169 		}
170 		return row;
171 	},
172 
173 	/**
174 	 * This function adds a column to a table and if obj is specified the column will be appended too.
175 	 * @param {object} obj The row/tr tag you wish to add the td/column too
176 	 * @returns {object} The jquery object of the newly created column
177 	 * @since 1.0
178 	 * @example
179 	 * provider.addColumn(container);
180 	 */
181 	addColumn : function(obj) {
182 		var column = $("<td></td>");
183 		if (typeof obj == "object" && obj != null) {
184 			obj.append(column);
185 		}
186 		return column;
187 	}
188  }
