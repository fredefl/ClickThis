 34 $(document).ready(function () {
 35 	// Shorten title if on home
 36 	if (isOnPage('home.html')) {
 37 		shortenTitle();
 38 	}
 39 	// Add the about box
 40 	addAboutBox();
 41 });
 42 
 43 $(window).resize(function (e) {
 44 	// Shorten title if on home
 45 	if (isOnPage('home.html')) {
 46 		shortenTitle();
 47 	}
 48 	// Resize the buttons if the browser window resizes
 49 	if (isOnPage('multiplechoice.html') || isOnPage('singlechoice.html') || isOnPage('buttons.html')) {
 50 		buttonResizer.resizeButtons(document.body);
 51 	}
 52 });
 53 
 54 document.addEventListener("orientationChanged", function () {
 55 	// Shorten title if on home
 56 	if (isOnPage('home.html')) {
 57 		shortenTitle();
 58 	}
 59 	// Resize the buttons if the orientation changes
 60 	if (isOnPage('multiplechoice.html') || isOnPage('singlechoice.html') || isOnPage('buttons.html')) {
 61 		buttonResizer.resizeButtons(document.body);
 62 	}
 63 });
 64 
 65 // If the hash changes
 66 $(window).hashchange(function () {
 67 	var Hash,
 68 		page;
 69 	if (!isOnPage('multiplechoice.html')) {
 70 		Hash = location.hash;
 71 		if (Hash !== null && Hash !== undefined && Hash !== '') {
 72 			page = Hash.replace('#', '');
 73 			changePage(page);
 74 		}
 75 	}
 76 });
